Flow C: Agno Team Pattern

Coordinate mode: leader orchestrates

from agno.team import Team engagement = Agent( name="Engagement", tools=[get_hubspot_activity], instructions="Analyze HubSpot engagement signals.") news = Agent( name="News", tools=[search_company_news], instructions="Find buying signals in company news.") jobs = Agent( name="Jobs", tools=[check_job_changes], instructions="Detect leadership changes.") scorer = Agent( name="Scorer", instructions="Score 0-100 from all signals. Recommend action.") team = Team( name="Lead Signal Detection", mode="coordinate", members=[engagement, news, jobs, scorer], )
